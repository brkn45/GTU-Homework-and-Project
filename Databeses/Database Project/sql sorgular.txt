CREATE TABLE USER_ (
    ID INT AUTO_INCREMENT,
    USERNAME_ VARCHAR(50),
    PASSWORD_ VARCHAR(50),
    NAMESURNAME VARCHAR(100),
    EMAIL VARCHAR(100),
    GENDER VARCHAR(1),
    CREATEDDATE DATETIME,
    BIRTHDATE DATE,
    TELNR1 VARCHAR(15),
    TELNR2 VARCHAR(15),
    PRIMARY KEY (ID)
);


CREATE TABLE ADDRESS (
    ID INT AUTO_INCREMENT,
    COUNTRYID INT,
    CITYID INT,
    TOWNID INT,
    DISTRICTID INT,
    POSTALCODE VARCHAR(10),
    ADDRESSTEXT VARCHAR(250),
    USERID INT,
    PRIMARY KEY (ID)
);

CREATE TABLE COUNTRY (
    ID INT AUTO_INCREMENT,
    COUNTRY VARCHAR(100),
    PRIMARY KEY (ID)
);

CREATE TABLE TOWN (
    ID INT AUTO_INCREMENT,
    TOWN VARCHAR(100),
    PRIMARY KEY (ID)
);

CREATE TABLE DISTRICT (
    ID INT AUTO_INCREMENT,
    DISTRICT VARCHAR(100),
    PRIMARY KEY (ID)
);

CREATE TABLE CITY (
    ID INT AUTO_INCREMENT,
    CITY VARCHAR(100),
    PRIMARY KEY (ID)
);

CREATE TABLE ITEM (
    ID INT AUTO_INCREMENT,
    ITEMCODE VARCHAR(20),
    ITEMNAME VARCHAR(100),
    PRICE FLOAT,
    CATEGORY1 VARCHAR(50),
    CATEGORY2 VARCHAR(50),
    CATEGORY3 VARCHAR(50),
    PRIMARY KEY (ID)
);

CREATE TABLE BASKET (
    ID INT AUTO_INCREMENT,
    USERID INT,
    CREATEDDATE DATETIME,
    LASTMODIFIEDDATE DATETIME,
    ITEMCOUNT INT,
    TOTALPRICE FLOAT,
    STATUS_ INT,
    PRIMARY KEY (ID)
);

CREATE TABLE BASKETDETAIL (
    ID INT AUTO_INCREMENT,
    BASKETID INT,
    ITEMID INT,
    AMOUNT FLOAT,
    PRICE FLOAT,
    TOTALPRICE FLOAT,
    DATE_ DATETIME,
    PRIMARY KEY (ID)
);

CREATE TABLE PAYMENT (
    ID INT AUTO_INCREMENT,
    BASKETID INT,
    TOTALPRICE FLOAT,
    PAYMENTTYPE INT,
    DATE_ DATETIME,
    ISOK BIT,
    APPROVECODE VARCHAR(20),
    ERROR_ VARCHAR(1000),
    PRIMARY KEY (ID)
);


CREATE TABLE `ORDER` (
    ID INT AUTO_INCREMENT,
    USERID INT,
    BASKETID INT,
    CREATEDDATE DATETIME,
    ITEMCOUNT INT,
    TOTALPRICE FLOAT,
    STATUS_ INT,
    PRIMARY KEY (ID)
);

CREATE TABLE ORDERDETAIL (
    ID INT AUTO_INCREMENT,
    ORDERID INT,
    BASKETDETAILID INT,
    ITEMID INT,
    AMOUNT FLOAT,
    PRICE FLOAT,
    TOTALPRICE FLOAT,
    PRIMARY KEY (ID)
);

CREATE TABLE INVOICE (
    ID INT AUTO_INCREMENT,
    ORDERID INT,
    INVOICENO VARCHAR(50),
    DATE_ DATETIME,
    CARGOFICHENO VARCHAR(50),
    STATUS_ INT,
    PRIMARY KEY (ID)
);

CREATE TABLE INVOICEDETAIL (
    ID INT AUTO_INCREMENT,
    INVOICEID INT,
    ORDERDETAILID INT,
    ITEMID INT,
    PRICE FLOAT,
    AMOUNT FLOAT,
    PRIMARY KEY (ID)
);
